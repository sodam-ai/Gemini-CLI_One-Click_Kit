@echo off
setlocal EnableDelayedExpansion
chcp 949 >nul 2>&1

REM ============================================================
REM  Gemini CLI 원클릭 키트 v5.1 (초보자용 / 숫자 메뉴)
REM  - 더블클릭 -> 숫자 키 하나.
REM  - 인코딩: CP949(한글 Windows cmd 호환) + CRLF
REM  - v5.0 대개편: 2026-06-18 구글 로그인 무료경로 종료 반영.
REM    [2] 'API 키 설정' 추가(setx 영구 + set 즉시). 메뉴 8개로 재번호.
REM    [3]실행/[4]질문은 API 키 확인 후 안내. call gemini / --skip-trust 유지.
REM    [8]제거에 API 키 삭제 옵션 추가.
REM  - v5.1: 상태 감지로 '지금 누를 번호' 동적 안내(DETECT_NEXT).
REM    설치/키설정/준비완료 상태에 따라 [1]/[2]/[3]을 정확히 짚어줌.
REM  - 편집은 이 _build_kit.txt(UTF-8) 수정 후 CP949+CRLF로 .bat 재생성.
REM  - 한글깨짐/파서오류 있으면 UTF-8+chcp65001 방식으로 전환 권장
REM ============================================================

set "SCRIPT_DIR=%~dp0"
set "LOG_FILE=%SCRIPT_DIR%gemini_cli.log"
set "REPORT=%SCRIPT_DIR%gemini_jindan.txt"
set "PKG=@google/gemini-cli"
set "CONFIG_DIR=%USERPROFILE%\.gemini"

REM === node / gemini 명령 인식을 위해 PATH 보강 (항상) ===
set "PATH=%APPDATA%\npm;%ProgramFiles%\nodejs;%PATH%"

title Gemini CLI 원클릭 키트

REM ============================================================
REM  메인 메뉴
REM ============================================================
:MENU
cls
call :DETECT_NEXT
echo ============================================================
echo            Gemini CLI 원클릭 키트   (초보자용)
echo ============================================================
echo.
echo   구글의 무료 AI 'Gemini'를 내 컴퓨터에서 바로 쓰는 도구입니다.
echo.
echo   순서:  1 설치  -^>  2 API 키 설정  -^>  3 실행
echo.
echo   ^>^> 지금 누를 것:  !NEXTMSG!
echo.
echo     1.  설치          처음 한 번만        ^( 추천 1단계 ^)
echo     2.  API 키 설정   무료 키 한 번 저장  ^( 추천 2단계 ^)
echo     3.  실행          AI와 대화 시작      ^( 추천 3단계 ^)
echo     4.  한 줄 질문    대화창 없이 질문 하나만
echo     5.  사용법        따라하기 + 예시
echo     6.  자가진단      안 될 때 원인 찾기
echo     7.  업데이트      최신 버전으로
echo   ----------------------------------------------------------
echo     8.  제거          지우기 ^( 누르면 한 번 더 물어봐요 ^)
echo     0.  종료          창 닫기
echo.
echo ------------------------------------------------------------
echo   숫자 키 하나만 누르면 바로 됩니다.  ^( Enter 안 눌러도 돼요 ^)
choice /c 123456780 /n /m "  원하는 번호 키: "
set "SEL=%errorlevel%"
if "%SEL%"=="1" goto INSTALL
if "%SEL%"=="2" goto APIKEY
if "%SEL%"=="3" goto RUN
if "%SEL%"=="4" goto ASK
if "%SEL%"=="5" goto USAGE
if "%SEL%"=="6" goto DIAGNOSE
if "%SEL%"=="7" goto UPDATE
if "%SEL%"=="8" goto UNINSTALL
if "%SEL%"=="9" goto END
goto MENU

REM ============================================================
REM  [1] 설치
REM ============================================================
:INSTALL
cls
echo ===== 1. 설치 =====
echo.
echo   완전 무료입니다. 신용카드 등록이 필요 없습니다.
echo.
call :CHECK_NODE
if errorlevel 1 (
    call :INSTALL_NODE
    if errorlevel 1 (
        echo.
        pause
        goto MENU
    )
)
echo   Node.js !NODE_VER! 확인됨.
echo.
where gemini >nul 2>&1
if not errorlevel 1 (
    echo   이미 설치되어 있습니다. 다음은 메뉴 2 API 키 설정으로 가세요.
    echo.
    pause
    goto MENU
)
echo   인터넷에서 Gemini AI를 내려받아 설치합니다. 보통 30초~1분.
echo   창이 멈춘 것처럼 보여도 점이 움직이는 동안 기다리세요.
echo.
call npm install -g %PKG%
if errorlevel 1 (
    echo.
    echo   설치 실패. 인터넷 연결을 확인하고 다시 시도하세요.
    echo   원인이 궁금하면 메뉴 6 자가진단을 눌러 보세요.
    echo   수동 설치: npm install -g %PKG%
    call :LOG "INSTALL failed"
    echo.
    pause
    goto MENU
)
set "PATH=%APPDATA%\npm;%PATH%"
echo.
echo   설치 완료! 다음은 메뉴 2 API 키 설정으로 무료 키를 넣으세요.
call :LOG "INSTALL success"
echo.
pause
goto MENU

REM ============================================================
REM  [2] API 키 설정 (무료 키 저장: setx 영구 + set 즉시)
REM ============================================================
:APIKEY
cls
echo ===== 2. API 키 설정 =====
echo.
echo   무료 Gemini를 쓰려면 'API 키'가 필요합니다.
echo   ^(2026-06-18부터 '구글 로그인' 무료 방식은 종료되었습니다^)
echo.
echo   [1단계] 무료 키 만들기  ^(신용카드 필요 없음^)
echo      1. 브라우저에서 https://aistudio.google.com/apikey 접속
echo      2. 구글 로그인 후 [Create API key] / [API 키 만들기] 클릭
echo      3. 만들어진 키 옆 '복사' 버튼 클릭  ^(AIza... 로 시작^)
echo.
echo   * 키는 비밀번호와 같습니다. 남에게 보여주지 마세요.
echo   * 무료 등급에는 개인정보/회사 자료를 입력하지 마세요.
echo.
echo   [2단계] 아래에 키를 붙여넣기  ^(마우스 오른쪽 클릭 = 붙여넣기^)
echo.
set "APIKEY_IN="
set /p "APIKEY_IN=키를 붙여넣고 Enter ^(그냥 Enter 치면 취소^): "
if not defined APIKEY_IN (
    echo.
    echo   입력이 없어 메뉴로 돌아갑니다.
    pause
    goto MENU
)
set "GEMINI_API_KEY=!APIKEY_IN!"
setx GEMINI_API_KEY "!APIKEY_IN!" >nul 2>&1
set "APIKEY_IN="
echo.
echo   저장 완료! 이제 메뉴 3 실행 / 4 한 줄 질문이 작동합니다.
echo   ^(키는 이 PC의 사용자 환경변수 GEMINI_API_KEY 에 저장됩니다^)
echo   ^(키를 지우려면 메뉴 8 제거에서 'API 키도 삭제'를 고르세요^)
call :LOG "APIKEY set"
echo.
pause
goto MENU

REM ============================================================
REM  [3] 실행
REM ============================================================
:RUN
cls
echo ===== 3. 실행 =====
echo.
call :CHECK_NODE
if errorlevel 1 (
    echo.
    echo   먼저 메뉴 1 설치를 눌러 Node.js와 Gemini를 준비하세요.
    pause
    goto MENU
)
where gemini >nul 2>&1
if errorlevel 1 (
    echo   아직 설치되지 않았습니다. 먼저 메뉴 1 설치를 눌러주세요.
    echo.
    pause
    goto MENU
)
if not defined GEMINI_API_KEY if not defined GOOGLE_API_KEY (
    echo   아직 API 키가 없습니다. 먼저 메뉴 2 API 키 설정을 해주세요.
    echo   ^(키를 방금 다른 창에서 넣었다면, 이 창을 닫고 다시 여세요^)
    echo.
    pause
    goto MENU
)
echo   준비됐습니다. 대화를 시작합니다.
echo.
echo   * 처음에 '이 폴더를 신뢰하시겠습니까?(trust)' 물으면 신뢰/Yes를 고르세요.
echo   * 대화 끝낼 때: 입력창에 /quit  또는 Ctrl+C 두 번
echo.
pause
cls
call :LOG "RUN gemini"
call gemini
echo.
echo ------------------------------------------------------------
echo   대화를 종료했습니다. 메뉴로 돌아갑니다.
call :LOG "RUN ended"
pause
goto MENU

REM ============================================================
REM  [4] 한 줄 질문 (대화창 없이 1회 질문)
REM ============================================================
:ASK
cls
echo ===== 4. 한 줄 질문 =====
echo.
echo   대화창에 들어가지 않고 질문 하나만 빠르게 물어봅니다.
echo.
where gemini >nul 2>&1
if errorlevel 1 (
    echo   아직 설치되지 않았습니다. 먼저 메뉴 1 설치를 눌러주세요.
    echo.
    pause
    goto MENU
)
if not defined GEMINI_API_KEY if not defined GOOGLE_API_KEY (
    echo   아직 API 키가 없습니다. 먼저 메뉴 2 API 키 설정을 해주세요.
    echo.
    pause
    goto MENU
)
echo   예 ^) 오늘 점심 메뉴 추천해줘   /   이메일 정중하게 써줘
echo.
set "Q="
set /p "Q=질문을 입력하고 Enter: "
if not defined Q (
    echo   입력이 없어 메뉴로 돌아갑니다.
    pause
    goto MENU
)
echo.
echo   답변을 만드는 중입니다... 몇 초 걸릴 수 있어요.
echo ------------------------------------------------------------
call :LOG "ASK one-line"
call gemini --skip-trust -p "!Q!"
echo ------------------------------------------------------------
echo.
pause
goto MENU

REM ============================================================
REM  [5] 사용법
REM ============================================================
:USAGE
cls
echo ===== 5. 사용법 =====
echo.
echo   * 가장 쉬운 순서
echo      1. 메뉴 1 설치 -^> 2 API 키 설정 -^> 3 실행 을 차례로.
echo      2. 대화창에 그냥 한국어로 적는다. 외울 명령어 없음.
echo      3. 답이 온다. 계속 이어서 물어보면 된다.
echo.
echo   * 이런 걸 물어보세요 ^(그대로 복사해서 써도 됩니다^)
echo      - 오늘 점심 메뉴 3개만 추천해줘
echo      - 아래 문장 맞춤법 고쳐줘: 그다음 문장 붙여넣기
echo      - 정중한 사과 이메일 써줘
echo      - 엑셀에서 합계 구하는 방법 알려줘
echo      - 파이썬으로 구구단 코드 짜줘
echo      - 이 영어 문장 한국어로 번역해줘
echo.
echo   * 대화창 안에서 쓰는 명령
echo      /help   도움말 보기
echo      /clear  화면 정리   또는 Ctrl+L
echo      /quit   대화 끝내기   또는 Ctrl+C 두 번
echo.
echo   * 대화창 없이 질문 하나만 -^> 메뉴 4 한 줄 질문
echo   * 잘 안 될 때 -^> 메뉴 6 자가진단
echo.
echo   * 무료 한도는 Google 기준이며 변동될 수 있습니다.
echo     ^(개인 무료 API 키, 자세한 한도는 aistudio.google.com 에서 확인^)
echo.
pause
goto MENU

REM ============================================================
REM  [6] 자가진단
REM ============================================================
:DIAGNOSE
cls
echo ===== 6. 자가진단 =====
echo.
set "D_NODE=없음"
set "D_NPM=없음"
set "D_NET=확인안됨"
set "D_GEM=없음"
set "D_GVER=-"
set "D_KEY=없음"
where node >nul 2>&1
if not errorlevel 1 (
    for /f "tokens=1" %%v in ('node --version') do set "D_NODE=%%v"
)
where npm >nul 2>&1
if not errorlevel 1 set "D_NPM=있음"
ping -n 1 registry.npmjs.org >nul 2>&1
if not errorlevel 1 set "D_NET=정상"
where gemini >nul 2>&1
if not errorlevel 1 (
    set "D_GEM=설치됨"
    for /f "delims=" %%v in ('gemini --version 2^>nul') do set "D_GVER=%%v"
)
if defined GEMINI_API_KEY set "D_KEY=있음"
if defined GOOGLE_API_KEY set "D_KEY=있음"
echo   Node.js 필수    : !D_NODE!
echo   npm 필수        : !D_NPM!
echo   인터넷 연결     : !D_NET!
echo   Gemini AI       : !D_GEM!   버전: !D_GVER!
echo   API 키 설정     : !D_KEY!
echo   설정폴더 위치   : %CONFIG_DIR%
echo.
(
echo Gemini CLI 진단 결과
echo 시간: %date% %time%
echo ----------------------------------------
echo Node.js     : !D_NODE!
echo npm         : !D_NPM!
echo 인터넷      : !D_NET!
echo Gemini AI   : !D_GEM!  버전 !D_GVER!
echo API 키      : !D_KEY!
echo 설정폴더    : %CONFIG_DIR%
) > "%REPORT%"
echo   이 내용을 파일로도 저장했습니다:
echo     %REPORT%
echo.
echo   ---- 이렇게 하세요 ----
set "FIXED=0"
if "!D_NET!"=="확인안됨" (
    echo   - 인터넷이 안 됩니다. 와이파이/랜선을 확인하세요.
    set "FIXED=1"
)
if "!D_NODE!"=="없음" (
    echo   - Node.js가 없습니다. 메뉴 1 설치를 누르면 자동으로 깔 수 있어요.
    set "FIXED=1"
)
if not "!D_NODE!"=="없음" if "!D_GEM!"=="없음" (
    echo   - Gemini가 없습니다. 메뉴 1 설치를 누르세요.
    set "FIXED=1"
)
if "!D_GEM!"=="설치됨" if "!D_KEY!"=="없음" (
    echo   - 아직 API 키가 없습니다. 메뉴 2 API 키 설정을 하세요.
    set "FIXED=1"
)
if "!FIXED!"=="0" echo   - 모두 정상입니다. 메뉴 3 실행을 누르면 바로 쓸 수 있어요.
call :LOG "DIAGNOSE done"
echo.
pause
goto MENU

REM ============================================================
REM  [7] 업데이트
REM ============================================================
:UPDATE
cls
echo ===== 7. 업데이트 =====
echo.
call :CHECK_NODE
if errorlevel 1 (
    echo.
    echo   먼저 메뉴 1 설치가 필요합니다.
    pause
    goto MENU
)
where gemini >nul 2>&1
if errorlevel 1 (
    echo   아직 설치되지 않았습니다. 메뉴 1 설치를 먼저 누르세요.
    echo.
    pause
    goto MENU
)
echo   최신 버전을 내려받아 적용합니다. 보통 30초~1분.
echo.
call npm install -g %PKG%@latest
if errorlevel 1 (
    echo.
    echo   업데이트 실패. 인터넷 연결을 확인하고 다시 시도하세요.
    echo   원인이 궁금하면 메뉴 6 자가진단을 눌러 보세요.
    call :LOG "UPDATE failed"
    echo.
    pause
    goto MENU
)
echo.
for /f "delims=" %%v in ('gemini --version 2^>nul') do set "GEMINI_VER=%%v"
echo   업데이트 완료! 현재 버전: !GEMINI_VER!
call :LOG "UPDATE success"
echo.
pause
goto MENU

REM ============================================================
REM  [8] 제거  (확인 후 진행)
REM ============================================================
:UNINSTALL
cls
echo ===== 8. 제거 =====
echo.
echo   Gemini AI를 컴퓨터에서 지웁니다.
echo   나중에 메뉴 1 설치로 다시 깔 수 있습니다.
echo.
choice /c YN /n /m "  정말 제거할까요?  Y=예 / N=아니오 : "
if errorlevel 2 goto MENU
echo.
echo   제거 중...
call npm uninstall -g %PKG%
if errorlevel 1 (
    echo   이미 지워져 있거나 문제가 있었습니다.
) else (
    echo   Gemini AI 프로그램을 제거했습니다.
)
call :LOG "UNINSTALL package"
echo.
echo   저장한 API 키도 지울까요?  ^(GEMINI_API_KEY 환경변수^)
choice /c YN /n /m "  API 키도 삭제할까요?  Y=예 / N=아니오 : "
if errorlevel 2 goto UNINSTALL_CFG
reg delete "HKCU\Environment" /v GEMINI_API_KEY /f >nul 2>&1
set "GEMINI_API_KEY="
echo   저장한 API 키를 삭제했습니다.
call :LOG "UNINSTALL apikey removed"
:UNINSTALL_CFG
echo.
echo   로그인 정보 등 설정 폴더도 지울까요?
echo   위치: %CONFIG_DIR%
choice /c YN /n /m "  설정 폴더도 지울까요?  Y=예 / N=아니오 : "
if errorlevel 2 goto UNINSTALL_DONE
if exist "%CONFIG_DIR%" (
    rd /s /q "%CONFIG_DIR%"
    echo   설정 폴더도 지웠습니다.
    call :LOG "UNINSTALL config removed"
) else (
    echo   설정 폴더는 이미 없습니다.
)
:UNINSTALL_DONE
echo.
pause
goto MENU

REM ============================================================
REM  보조 함수: Node.js 자동 설치 (winget 우선, 실패 시 수동 안내)
REM ============================================================
:INSTALL_NODE
echo.
echo   Gemini AI를 쓰려면 'Node.js'라는 무료 기반 프로그램이 필요합니다.
echo.
where winget >nul 2>&1
if errorlevel 1 (
    echo   이 컴퓨터에서는 자동 설치 도구 winget 을 찾지 못했습니다.
    echo   아래 주소에서 직접 설치해 주세요. 왼쪽 'LTS' 버튼입니다.
    echo       https://nodejs.org
    echo   설치 후 이 창을 닫았다가 다시 실행하세요.
    call :LOG "INSTALL_NODE: winget not found"
    exit /b 1
)
echo   자동으로 설치할 수 있습니다. winget 사용, 보통 1~3분.
choice /c YN /n /m "  지금 자동 설치할까요?  Y=예 / N=아니오 : "
if errorlevel 2 (
    echo   취소했습니다. 직접 설치하려면 https://nodejs.org 의 LTS 버튼을 쓰세요.
    exit /b 1
)
echo.
echo   Node.js 설치 중... 창이 멈춘 듯 보여도 기다리세요.
winget install -e --id OpenJS.NodeJS.LTS --accept-source-agreements --accept-package-agreements
set "PATH=%ProgramFiles%\nodejs;%APPDATA%\npm;%PATH%"
where node >nul 2>&1
if errorlevel 1 (
    echo.
    echo   설치는 됐지만 이 창에서 아직 인식되지 않습니다.
    echo   이 창을 닫았다가 키트를 다시 실행해 주세요.
    echo   그 다음 [1] 설치를 누르면 됩니다.
    call :LOG "INSTALL_NODE: installed, needs reopen"
    exit /b 1
)
for /f "tokens=1" %%v in ('node --version') do set "NODE_VER=%%v"
set "NODE_VER=!NODE_VER:v=!"
echo   Node.js !NODE_VER! 설치 완료.
call :LOG "INSTALL_NODE: success !NODE_VER!"
exit /b 0

REM ============================================================
REM  보조 함수: Node.js 확인 (없거나 20 미만이면 errorlevel 1)
REM ============================================================
:CHECK_NODE
where node >nul 2>&1
if errorlevel 1 (
    echo   [주의] Node.js가 설치되어 있지 않습니다.
    call :LOG "CHECK_NODE: node not found"
    exit /b 1
)
for /f "tokens=1" %%v in ('node --version') do set "NODE_VER=%%v"
set "NODE_VER=!NODE_VER:v=!"
for /f "tokens=1 delims=." %%a in ("!NODE_VER!") do set "NODE_MAJOR=%%a"
if !NODE_MAJOR! LSS 20 (
    echo   [주의] Node.js 버전이 낮습니다. 현재 !NODE_VER!, 필요 20 이상.
    echo          https://nodejs.org 에서 'LTS'로 업데이트하세요.
    call :LOG "CHECK_NODE: node too old !NODE_VER!"
    exit /b 1
)
exit /b 0

REM ============================================================
REM  보조 함수: 상태 감지로 '지금 누를 번호' 안내
REM ============================================================
:DETECT_NEXT

if defined GOOGLE_API_KEY set "NEXTMSG=준비 완료. 이제  [3] 실행  을 누르세요"
exit /b

:LOG

exit /b 0

:END
